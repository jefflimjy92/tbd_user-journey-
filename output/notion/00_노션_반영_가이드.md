```bash
python3 - <<'PY'
import os,re,glob
base='/Users/joonyounglim/Desktop/CSO_전략기획/output/notion'
files=sorted(glob.glob(base+'/0[1-9]_*.md')+glob.glob(base+'/1[0-3]_*.md'))
exists={os.path.basename(f) for f in files}
for f in files:
    txt=open(f,'r',encoding='utf-8').read()
    links=re.findall(r'\((\./[^)]+\.md)\)',txt)
    miss=[os.path.basename(l) for l in links if os.path.basename(l) not in exists]

PY
```
